async function runAnalysis(){ if(!selDay || !days[selDay]) return; setAnalysis(null); setAnalysing(true); setModal("analysis"); const es = days[selDay] || []; const dayEs = es.filter(e=>!e.night).sort((a,b)=>timeVal(a)-timeVal(b)); const nightEs = es.filter(e=>e.night); const feedEs = es.filter(e=>e.type==="feed"); const napEs = dayEs.filter(e=>e.type==="nap"); const wake = dayEs.find(e=>e.type==="wake"); const sleep = dayEs.find(e=>e.type==="sleep"); const totalMlDay = feedEs.reduce((s,f)=>s+(f.amount||0),0); const totalNapM = napEs.reduce((s,n)=>s+minDiff(n.start,n.end),0); const prompt = ` You are a warm baby sleep consultant. Date: ${fmtLong(selDay)} Baby: ${babyName || "Baby"} Wake: ${wake ? fmt12(wake.time) : "not logged"} Feeds: ${feedEs.map(f => fmt12(f.time) + " (" + f.amount + "ml)").join(", ")} Naps: ${napEs.map(n => fmt12(n.start) + "-" + fmt12(n.end) + " (" + minDiff(n.start,n.end) + "min)").join(", ")} Bedtime: ${sleep ? fmt12(sleep.time) : "not logged"} Night wakes: ${nightEs.map(e => fmt12(e.time)).join(", ")} Total feed: ${totalMlDay}ml Total nap: ${hm(totalNapM)} Please provide: 1. What went well 2. Sleep observations 3. Suggestions for tomorrow 4. Encouraging note `; try{ const res = await fetch(WORKER_URL,{ method:"POST", headers:{ "Content-Type":"application/json" }, body: JSON.stringify({ model:"gpt-4o-mini", messages:[ {role:"system",content:"You are a helpful baby sleep consultant."}, {role:"user",content:prompt} ] }) }); const data = await res.json(); const full = data.choices?.[0]?.message?.content || "No analysis returned."; setAnalysis(full); }catch{ setAnalysis("Sorry, couldn't connect to the AI service."); } setAnalysing(false); }
